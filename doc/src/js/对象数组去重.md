# 对象数组去重

```js
// 去重如下数组，
var arr = [
    {a:1, b:2}, {a:1, b:2},{a:1,b:2,c:3},{a:1,b:2,c:{d:4}}
]
const result = []; // 去重后的结果
for (const item1 of arr) {
    let isFind = false;
    for (const item2 of result) {
        if(equal(item1, item2)) {
            isFind = true;
            break;
        }
    }
    if (!isFind) {
        result.push(item1);
    }
}

// 中间函数，用于确认两对象是否相等（对象的健值对相等）
const equal = (a, b) => {
    if (a === b) return true;
    if (!isObject(a) || !isObject(b)) {
        return Object.is(a, b);
    };
    if (Object.keys(a).length !== Object.keys(b).length) return false;
    for (const key in a) {
        if (!equal(a[key], b[key])) return false;
    }
    return true;
}
const isObject = (obj) => {
    return obj !== null && typeof obj === 'object';
}
```
